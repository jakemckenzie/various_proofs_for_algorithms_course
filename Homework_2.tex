\begin{algorithm}
\caption{Find Max integer in an Array with simple recursion}
\label{array-sum}
\begin{algorithmic}[1]
\Procedure{Find Max}{$A, index$}
    \If{($index == A.Length - 1$)}
        \State return A[A.Length - 1]
    \EndIf
    \State return Find Max(A[index], Find Max(A[index + 1]))
\EndProcedure
\end{algorithmic}
\end{algorithm}
