\begin{algorithm}
\caption{Find Max integer in an Array with simple recursion}
\label{array-sum}
\begin{algorithmic}[1]
\Procedure{Find Max}{$A$}
    \If{($a == b$)}

\[

\begin{enumerate}

$$n = 1; T(1) = 1$$

\[
  T(k) =
